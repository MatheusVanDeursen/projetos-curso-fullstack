--CRIAÇÃO DE TABLES--

CREATE TABLE tab_alunos (
    id SMALLINT PRIMARY KEY AUTO_INCREMENT,
    nome varchar(50) NOT NULL,
    data_nascimento DATE NOT NULL,
    telefone varchar(15),
    endereco varchar(100) NOT NULL,
    responsavel_financeiro varchar(50) NOT NULL
);

CREATE TABLE tab_professores (
    id SMALLINT PRIMARY KEY AUTO_INCREMENT,
    nome varchar(50) NOT NULL,
    data_nascimento DATE NOT NULL,
    telefone varchar(15) NOT NULL,
    email varchar(64) NOT NULL,
    endereco varchar(100) NOT NULL,
    disciplina varchar(30)
);

CREATE TABLE tab_cursos (
    id SMALLINT PRIMARY KEY AUTO_INCREMENT,
    nome varchar(10),
    preco float
);

CREATE TABLE tab_turmas (
    id SMALLINT PRIMARY KEY AUTO_INCREMENT,
    nome varchar(3) NOT NULL,
    professor_id SMALLINT,
    hora_inicio time,
    hora_fim time,
    curso_id SMALLINT NOT NULL,
    
    CONSTRAINT FOREIGN KEY FK_TURMAS_CURSO
	(curso_id) REFERENCES tab_cursos(id),
    CONSTRAINT FOREIGN KEY FK_TURMAS_PROFESSOR
	(professor_id) REFERENCES tab_professores(id)
);

CREATE TABLE tab_matriculas (
    id SMALLINT PRIMARY KEY AUTO_INCREMENT,
    codigo_matricula varchar(32),
    data_matriculado date DEFAULT now(),
    aluno_id SMALLINT NOT NULL,
    turma_id SMALLINT,
    
    CONSTRAINT FOREIGN KEY FK_MATRICULAS_ALUNO
	(aluno_id) REFERENCES tab_alunos(id),
    CONSTRAINT FOREIGN KEY FK_MATRICULAS_TURMA
	(turma_id) REFERENCES tab_turmas(id)
);
